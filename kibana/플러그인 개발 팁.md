## 로그인 세션 정보 가져오기
- router 에서 아래와 같이 가져올 수 있음
    ``` javascript
    export default function (server) {
    
      server.route({
        path: '/api/kgtest/example',
        method: 'GET',
        handler(req, res) {
          //유저 정보 가져오기, 아이디, 설정 권한, 등 조회 가능
          server.plugins.security.getUser(req).then(user => {
            console.log(user.username);
            console.log(user);
          });
          ....
        }
      });
    }
    ```
